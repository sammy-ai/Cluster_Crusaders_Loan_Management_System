#create a keyspace named loan_management

CREATE KEYSPACE loan_management
  WITH REPLICATION = {
    'class' : 'NetworkTopologyStrategy',
    'datacenter1': 3
  };


#Create tables

USE loan_management;

CREATE TABLE Loans (
   L_ID int,
   B_ID int,
   agent text,
   amount double,
   loan_term text,
   date timestamp,
   status text,
   interest_rate double,
   balance double,
   branch text,
   PRIMARY KEY ( L_ID, date)) WITH CLUSTERING ORDER BY (date DESC);

CREATE TABLE Borrowers (
   B_ID int,
   names text,
   contact text,
   address text,
   employed boolean,
   monthly_income double,
   PRIMARY KEY (B_ID, names)
) WITH CLUSTERING ORDER BY (names ASC);

CREATE TABLE Agents (
   A_ID int,
   Br_ID int,
   names text,
   email text,
   phone_number text,
   region text,
   PRIMARY KEY (A_ID, names)
)  WITH CLUSTERING ORDER BY (names ASC);

CREATE TABLE Payments (
   P_ID int,
   L_ID int,
   amount double,
   date timestamp,
   charge double,
   PRIMARY KEY (P_ID, date)
) WITH CLUSTERING ORDER BY (date DESC);

CREATE TABLE Collateral (
   L_ID int,
   B_ID int,
   collateral_type text,
   collateral_value text,
   description text,
    PRIMARY KEY (B_ID, L_ID)
);

CREATE TABLE Branches (
   Br_ID int,
   br_name text,
   address text,
   contact text,
   region text,
   PRIMARY KEY (Br_ID)
);
